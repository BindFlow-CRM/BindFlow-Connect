SCHEMA COMPLETO:

-- ORGANIZATIONS (workspaces/agencias)
CREATE TABLE organizations (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  owner_id UUID REFERENCES auth.users(id),
  plan TEXT DEFAULT 'trial',
  trial_ends_at TIMESTAMPTZ DEFAULT NOW() + INTERVAL '14 days',
  subscription_id TEXT,
  subscription_status TEXT DEFAULT 'trialing',
  max_seats INT DEFAULT 3,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- ORGANIZATION MEMBERS
CREATE TABLE organization_members (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  user_id UUID REFERENCES auth.users(id) ON DELETE CASCADE,
  role TEXT DEFAULT 'member', -- 'owner', 'admin', 'member'
  invited_email TEXT,
  status TEXT DEFAULT 'active', -- 'active', 'invited', 'suspended'
  joined_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(organization_id, user_id)
);

-- PROFILES (extensión de auth.users)
CREATE TABLE profiles (
  id UUID PRIMARY KEY REFERENCES auth.users(id) ON DELETE CASCADE,
  full_name TEXT,
  avatar_url TEXT,
  phone TEXT,
  state TEXT, -- estado de EEUU
  license_number TEXT, -- número de licencia de agente
  agency_name TEXT,
  current_organization_id UUID REFERENCES organizations(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- PIPELINE STAGES
CREATE TABLE pipeline_stages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  color TEXT NOT NULL,
  position INT NOT NULL,
  is_default BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- CONTACTS (clientes del agente)
CREATE TABLE contacts (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  created_by UUID REFERENCES auth.users(id),
  assigned_to UUID REFERENCES auth.users(id),

  -- Datos personales
  full_name TEXT NOT NULL,
  email TEXT,
  phone TEXT,
  date_of_birth DATE,
  address TEXT,
  city TEXT,
  state TEXT,
  zip_code TEXT,

  -- Origen y clasificación
  lead_source TEXT, -- 'referral','web','cold_call','social','angi','google','other'
  status TEXT DEFAULT 'active', -- 'active','inactive','prospect'

  -- Metadata
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- POLICIES (pólizas por contacto)
CREATE TABLE policies (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  contact_id UUID REFERENCES contacts(id) ON DELETE CASCADE,
  pipeline_stage_id UUID REFERENCES pipeline_stages(id),

  -- Datos de póliza
  policy_number TEXT,
  insurance_company TEXT,
  line_of_insurance TEXT, -- 'auto','home','life','health','commercial','medicare','other'
  annual_premium DECIMAL(10,2),
  start_date DATE,
  renewal_date DATE,
  policy_status TEXT DEFAULT 'active', -- 'active','lapsed','cancelled','pending','quoted'

  -- Cross-sell
  current_products TEXT[], -- array de productos actuales
  cross_sell_opportunities TEXT[], -- array de oportunidades

  -- Metadata
  notes TEXT,
  created_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- DEALS (cards en el pipeline)
CREATE TABLE deals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  contact_id UUID REFERENCES contacts(id) ON DELETE CASCADE,
  policy_id UUID REFERENCES policies(id),
  stage_id UUID REFERENCES pipeline_stages(id),
  assigned_to UUID REFERENCES auth.users(id),

  title TEXT NOT NULL,
  value DECIMAL(10,2),
  expected_close_date DATE,
  position INT DEFAULT 0,
  notes TEXT,

  created_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- ACTIVITIES (timeline de actividades)
CREATE TABLE activities (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  contact_id UUID REFERENCES contacts(id) ON DELETE CASCADE,
  deal_id UUID REFERENCES deals(id),
  created_by UUID REFERENCES auth.users(id),

  type TEXT NOT NULL, -- 'note','call','email','meeting','whatsapp','stage_change','reminder_sent'
  title TEXT,
  content TEXT,
  metadata JSONB,

  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- REMINDERS (recordatorios)
CREATE TABLE reminders (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  contact_id UUID REFERENCES contacts(id),
  deal_id UUID REFERENCES deals(id),
  policy_id UUID REFERENCES policies(id),
  assigned_to UUID REFERENCES auth.users(id),

  title TEXT NOT NULL,
  notes TEXT,
  due_date TIMESTAMPTZ NOT NULL,
  reminder_type TEXT DEFAULT 'manual', -- 'manual','renewal_90','renewal_60','renewal_30'
  status TEXT DEFAULT 'pending', -- 'pending','completed','dismissed'
  is_sent BOOLEAN DEFAULT false,

  created_by UUID REFERENCES auth.users(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- TAGS
CREATE TABLE tags (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  color TEXT DEFAULT '#00E5A0',
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(organization_id, name)
);

-- CONTACT TAGS (relación many-to-many)
CREATE TABLE contact_tags (
  contact_id UUID REFERENCES contacts(id) ON DELETE CASCADE,
  tag_id UUID REFERENCES tags(id) ON DELETE CASCADE,
  PRIMARY KEY (contact_id, tag_id)
);

-- REFERRALS (tracker de referidos)
CREATE TABLE referrals (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  referrer_contact_id UUID REFERENCES contacts(id),
  referred_contact_id UUID REFERENCES contacts(id),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- EMAIL TEMPLATES
CREATE TABLE email_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  organization_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
  name TEXT NOT NULL,
  subject TEXT NOT NULL,
  body TEXT NOT NULL,
  template_type TEXT, -- 'renewal_90','renewal_60','renewal_30','follow_up','custom'
  is_default BOOLEAN DEFAULT false,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

PLAN PREMIUM (único plan):
→ Precio mensual:  $39 USD/mes
→ Precio anual:    $397 USD/año (15% descuento, ~$33/mes)
→ Seats incluidos: 3 usuarios por workspace
→ Trial:           14 días gratis, sin tarjeta de crédito
→ Procesador:      Paddle (Merchant of Record)
→ Al expirar trial: acceso bloqueado, datos conservados 30 días

LÓGICA DE TRIAL:
→ Registro → trial_ends_at = NOW() + 14 days
→ Durante trial: acceso completo sin restricción
→ Día 10: email recordatorio "4 días para que expire tu trial"
→ Día 13: email urgencia "Mañana expira tu trial"
→ Día 14: bloqueo de acceso → redirect a /billing
→ Datos conservados 30 días post-expiración
→ Cancelación: datos conservados hasta fin del período pagado
